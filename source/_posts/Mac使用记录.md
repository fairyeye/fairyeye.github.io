return geocodes

def signature(data: dict):

keys = sorted(data.keys())

temp_v = ""

for item in keys:

temp_v += data[item]

text = SALT + temp_v + CURRENT_TIME

hl = hashlib.md5()

hl.update(text.encode(encoding="utf8"))

md5 = hl.hexdigest()

return md5
