GROUP BY

```
Map<String, List<WmsIqcRecord>> map = records.stream().collect(Collectors.groupingBy(iqcRecord -> {
                    if (StringUtils.equals("TO_DO", iqcRecord.getStatus())) {
                        switch (iqcRecord.getDealMethod()) {
                            case "RELEASE":
                                return "TO_RELEASE";
                            case "FREEZE":
                                return "TO_FREEZE";
                            case "RETURN":
                                return "TO_RETURN";
                            default:
                                return "TO_DO";
                        }
                    } else if (StringUtils.equals("DONE", iqcRecord.getStatus())) {
                        switch (iqcRecord.getDealMethod()) {
                            case "RELEASE":
                                return "RELEASE_DONE";
                            case "FREEZE":
                                return "FREEZE_DONE";
                            case "RETURN":
                                return "RETURN_DONE";
                            default:
                                return "DONE";
                        }
                    } else {
                        return "DEAL";
                    }
                }));
```
